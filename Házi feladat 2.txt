CREATE TABLE dbo.Vendeg_Maszk (
    VEND_ID INT NOT NULL PRIMARY KEY,
    NEV NVARCHAR(100) MASKED WITH (FUNCTION = 'partial(1, "XXXX", 1)') NOT NULL,
    EMAIL NVARCHAR(100) MASKED WITH (FUNCTION = 'email()') NULL,
    SZUL_DAT DATE MASKED WITH (FUNCTION = 'default()') NULL,
  	SZAML_CIM NVARCHAR(100) MASKED WITH (FUNCTION = 'partial(5, "XXX", 5)'),
);

INSERT INTO dbo.Vendeg_Maszk (VEND_ID, NEV, EMAIL, SZUL_DAT, TELEFON, VAROS)
SELECT VEND_ID, NEV, EMAIL, SZUL_DAT, TELEFON, VAROS
FROM dbo.Vendeg;

CREATE USER MaszkFelhasznalo WITHOUT LOGIN;
GRANT SELECT ON dbo.Vendeg_Maszk TO MaszkFelhasznalo;
